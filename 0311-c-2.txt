#include <stdio.h>

int main() 
{
    int num;
    printf("숫자를 입력하세요:");
    scanf("%d", &num);
    //조건문
    
    if(num % 2 == 0 ) //if는 조건을 물어봄 무조건 소괄호를 써야함
    //%는 산술연산자로 나머지를 의미함
    { //실행해야하는 조건이 2개라서 중괄호로 나눠줌
    printf("짝수\n");
    printf("2로 나눈 나머지가 0\n");
}
    // 참일때만 수행,거짓일때는 수행x
    // >초과, <미만, >=이상, <=이하
    //=대입,==같다, !=같지않다
    //대입연산자의 !는 부정을 이야기함
    
    else { //실행해야하는 조건이 2개라서 중괄호로 나눠줌
    printf("홀수\n");
    printf("2로 나눈 나머지가 1\n");
}
    return 0;
}
